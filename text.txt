# ‚úÖ –í–∞—Ä–∏–∞–Ω—Ç 2: Python-—Å–∫—Ä–∏–ø—Ç, –∫–æ—Ç–æ—Ä—ã–π –ø–µ—Ä–µ–Ω–æ—Å–∏—Ç —Ç–∞–±–ª–∏—Ü—ã –Ω–∞–ø—Ä—è–º—É—é

–ï—Å–ª–∏ —Ö–æ—á–µ—à—å –ø–µ—Ä–µ–Ω–æ—Å–∏—Ç—å —Ç–∞–±–ª–∏—Ü—ã –ø—Ä–æ–≥—Ä–∞–º–º–Ω–æ:

```python
import sqlite3
import mysql.connector

# --- –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ SQLite ---
sqlite_conn = sqlite3.connect("db.sqlite3")
sqlite_cursor = sqlite_conn.cursor()

# --- –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ MySQL ---
mysql_conn = mysql.connector.connect(
    host="localhost",
    user="root",
    password="password",
    database="mydb"
)
mysql_cursor = mysql_conn.cursor()

# –ø–æ–ª—É—á–∏—Ç—å —Å–ø–∏—Å–æ–∫ —Ç–∞–±–ª–∏—Ü SQLite
sqlite_cursor.execute("SELECT name FROM sqlite_master WHERE type='table';")
tables = sqlite_cursor.fetchall()

for table_name, in tables:
    # –ø—Ä–æ–ø—É—Å—Ç–∏—Ç—å —Å–ª—É–∂–µ–±–Ω—ã–µ —Ç–∞–±–ª–∏—Ü—ã SQLite
    if table_name == "sqlite_sequence":
        continue

    # –ø–æ–ª—É—á–∏—Ç—å —Å—Ö–µ–º—É —Ç–∞–±–ª–∏—Ü—ã
    sqlite_cursor.execute(f"PRAGMA table_info({table_name});")
    columns = sqlite_cursor.fetchall()

    # —Å–æ–±—Ä–∞—Ç—å SQL "CREATE TABLE" –¥–ª—è MySQL
    col_defs = []
    for col in columns:
        name = col[1]
        col_type = col[2].upper()

        # –∫–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è —Ç–∏–ø–æ–≤ SQLite ‚Üí MySQL
        if "INT" in col_type:
            col_type = "INT"
        elif "CHAR" in col_type or "TEXT" in col_type:
            col_type = "VARCHAR(255)"
        elif "REAL" in col_type or "FLOA" in col_type or "DOUB" in col_type:
            col_type = "DOUBLE"
        else:
            col_type = "TEXT"

        col_defs.append(f"`{name}` {col_type}")

    create_sql = f"CREATE TABLE IF NOT EXISTS `{table_name}` ({', '.join(col_defs)});"
    mysql_cursor.execute(create_sql)

    # –∑–∞–≥—Ä—É–∑–∏—Ç—å –¥–∞–Ω–Ω—ã–µ
    sqlite_cursor.execute(f"SELECT * FROM {table_name}")
    rows = sqlite_cursor.fetchall()

    placeholders = ",".join(["%s"] * len(columns))
    insert_sql = f"INSERT INTO `{table_name}` VALUES ({placeholders})"

    for row in rows:
        mysql_cursor.execute(insert_sql, row)

mysql_conn.commit()

print("–ü–µ—Ä–µ–Ω–æ—Å –∑–∞–≤–µ—Ä—à—ë–Ω!")
```

import sqlite3

def list_tables_sqlite(db_file):
    """
    Подключается к базе данных SQLite и выводит список всех пользовательских таблиц.
    """
    conn = None
    try:
        conn = sqlite3.connect(db_file)
        cursor = conn.cursor()
        
        print(f"Подключено к базе данных: {db_file}")

        # SQL-запрос для получения имен таблиц
        # sqlite_master — это системная таблица, хранящая метаданные БД.
        # type='table' отфильтровывает индексы и представления.
        # name NOT LIKE 'sqlite_%' исключает внутренние системные таблицы.
        cursor.execute("SELECT name FROM sqlite_master WHERE type='table' AND name NOT LIKE 'sqlite_%'")
        
        tables = cursor.fetchall()
        
        if tables:
            print("\nСписок таблиц в базе данных:")
            for table in tables:
                # fetchall() возвращает список кортежей [(table_name,), ...]
                print(f"- {table[0]}")
        else:
            print("\nВ базе данных нет пользовательских таблиц.")

    except sqlite3.Error as e:
        print(f"Произошла ошибка базы данных: {e}")
    finally:
        if conn:
            conn.close()
            print("Соединение с БД закрыто.")

# --- Пример использования ---
# Укажите путь к вашей базе данных
database_file_path = 'моя_база.db.sqlite3'

list_tables_sqlite(database_file_path)
